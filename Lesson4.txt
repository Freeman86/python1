def get_random_numbers(length):
    nums = []

    for i in range(length):
        nums.append(str(random.randint(0, 9)))

    if nums[0] == 0:
        nums[0] = str(random.randint(1, 9))

    return "".join(nums)


def get_longest_sequence(numbers):
    max_sequence = 0
    max_sequence_number = None
    last = 0
    count = 0
    for num in numbers:
        if last == num:
            count += 1
        else:
            count = 1

        if count > max_sequence:
            max_sequence = count
            max_sequence_number = last

        last = num

    return max_sequence, max_sequence_number

    f.write(nums)
